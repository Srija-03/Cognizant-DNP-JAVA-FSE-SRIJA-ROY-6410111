BEGIN
    FOR customer IN (
        SELECT CustomerID,
               TRUNC(MONTHS_BETWEEN(SYSDATE, DOB) / 12) AS age FROM Customers)
	LOOP
        IF customer.age > 60 THEN
            UPDATE Loans
            SET InterestRate = InterestRate - 1
            WHERE CustomerID = customer.CustomerID;
        END IF;
    END LOOP;
    COMMIT;
END;
/
